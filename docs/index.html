render GovukComponent::Accordion.new(id: 'abc123') do |component|
  component.slot(:section,
    title: 'Home electronics',
    summary: 'Entertainment, communication and recreation',
    expanded: true) do
      tag.p(class: 'govuk-body') do
        "Lorem ipsum dolor sit amet, consectetur adipiscing elit."
      end
    end

  component.slot(:section,
    title: 'Appliances', summary: 'Laundry, cookers and vacuum cleaners') do
      tag.p(class: 'govuk-body') do
        "Ut et elementum tortor. Donec a tellus sed sem euismod congue"
      end
  end

  component.slot(:section, title: 'Toys') do
    safe_join([
      tag.p(class: 'govuk-body') do
        "In et nunc vitae tortor sollicitudin ultrices. Vivamus a purus justo."
      end,

      tag.p(class: 'govuk-body') do
        "Duis a turpis non nunc pulvinar venenatis quis congue mauris"
      end
    ])
  end
end

Cookie banner

Cookie banner

Cookies on a service

A paragraph.

A second, smaller paragraph.

Accept cookiesReject cookiesView cookies 
render GovukComponent::CookieBanner.new(title: "Cookies on a service") do |component|
  component.body do
    safe_join([
      tag.p("A paragraph.", class: "govuk-body"),
      tag.p("A second, smaller paragraph.", class: "govuk-body-s"),
    ])
  end
  component.actions do
    safe_join([
      govuk_link_to("Accept cookies", "#accept", button: true),
      govuk_link_to("Reject cookies", "#reject", button: true),
      govuk_link_to("View cookies", "#view"),
    ])
  end
end
